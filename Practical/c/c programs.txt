    int a,b,c;
    printf("Enter first numbers: ");
    scanf("%d",&a);
    printf("Enter second numbers: ");
    scanf("%d",&b);
    printf("Enter third numbers: ");
    scanf("%d",&c);
    if(a > b && a > c){
        printf("%d is greater than all",a);
    } else if(b > a && b > c){
        printf("%d is greater than all",b);
    } else if(c > a && c > b){
        printf("%d is greater than all",c);
    } else if(a == b && b==c){
        printf("All are equale");
    }
    return 0;
}
